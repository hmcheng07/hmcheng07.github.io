    # 表空间

    --创建表空间
create tablespace helloworld --指定表空间名
datafile 'c:\helloworld.dbf' --指定表空间对应的数据文件
size 100m --表空间的初始大小
autoextend on --当表空间存储都占满时，自动增长
next 10m; --每次自动增长的大小

--删除表空间
drop tablespace helloworld;

--创建用户
create user helloworld
identified by helloworld --指定密码
default tablespace helloworld; --指定表空间

    # 查询

    分组查询：

    --查询出每个部门的平均工资
select e.deptno, avg(e.sal) from emp e group by e.deptno;

--查询出平均工资高于2000的部门信息
select e.deptno, avg(e.sal) from emp e group by e.deptno having avg(e.sal)>2000;

--查询出每个部门工资高于800的员工的平均工资
select e.deptno, avg(e.sal) from emp e where e.sal>800 group by e.deptno;

--查询出每个部门工资高于800的员工的平均工资,然后再查出平均工资高于2000的部门
select e.deptno, avg(e.sal) from emp e where e.sal>800 group by e.deptno having avg(e.sal)>2000;

--注意：所有条件都不能使用别名来判断(有子查询除外)

    # 索引

    ​ 索引就是在表的列上构建一个二叉树，达到大幅度提高查询效率的目的，但是索引会影响增删改查的效率

    1.单列索引（构建在单列上）

    创建单列索引：

    create index idx_ename on emp(ename);
    1

    单列索引触发规则，条件必须是索引列中的原始值：

    select * from emp where ename='SCOTT';
    1

    单行函数，模糊查询，都会影响索引的触发

    2.复合索引（构建在多列上）

    创建复合索引：

    create index idx_enamejob on emp(ename,job);
    1

    复合索引中第一列为优先检索列，如果要触发复合索引，必须包含有优先检索列中的原始值

    select * from emp where ename='SCOTT' and job='xx';	--触发复合索引
    1

    条件为or时不触发索引：

    select * from emp where ename='SCOTT' or job='xx';	--不触发索引
    1

    如果某一列既是单列索引又是复合索引，查询条件仅包含这一列时，触发的是单列索引

    select * from emp where ename='SCOTT';	--触发单列索引
    1

    in和out类型参数的区别是什么？

    • 凡是涉及到into查询语句赋值或者 := 赋值操作的参数，都必须使用out来修饰

    # 存储过程和存储函数的区别？

    • 存储函数比存储过程多了两个return
    • 本质区别：存储函数有返回值，存储过程没有返回值
    • 存储过程要想实现有返回值的业务，必须使用out类型的参数
    • 可以使用存储函数有返回值的特性，自定义函数；存储过程不能用来自定义函数

    # 触发器

    ​ 制定一个规则，在我们做增删改操作的时候，只要满足该规则，自动触发，无需调用

    触发器中触发语句与伪记录变量的值：

    触发语句 ：old ：new
    Insert 所有字段都是空（null） 将要插入的数据
    Update 更新以前该行的值 更新后的值
    Delete 删除以前该行的值 所有字段都是空（null）

    **语句级触发器：**不包含有for each row的就是语句级触发器

    --语句级触发器
--插入一条记录，输出一个新员工入职

after
insert
on person

begin

end;

--触发t1
